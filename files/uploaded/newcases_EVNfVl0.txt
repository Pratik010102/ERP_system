    // Function to populate the district dropdown based on the selected state
    function populateDistrictDropdown() {
        var stateDropdown = document.getElementById("stateDropdown");
        var selectedState = stateDropdown.value;
        var districtDropdown = document.getElementById("districtDropdown");
        districtDropdown.innerHTML = "<option>Please Select</option>";

        var districts = stateDistricts[selectedState];
        if (districts && districts.length > 0) {
            districts.forEach(function(district) {
                var option = document.createElement("option");
                option.text = district;
                option.value = district;
                districtDropdown.appendChild(option);
            });
            document.getElementById("districtDropdownContainer").style.display = "block";
        } else {
            document.getElementById("districtDropdownContainer").style.display = "none";
        }
    }
